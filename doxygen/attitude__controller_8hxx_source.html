00039  * \brief A controller for attitude control, to be use within a cascade controller structure
00040  *
00041  * TODO: update this text
00042  * \details It takes a command in attitude (roll/pitch/yaw or quaternion) as
00043  * input, and computes a torque command on roll pitch and yaw.
00044  * The inner PID loop controls the angular speed around the 3 axis. This inner
00045  * loop is fed by the outer PID loop which controls the attitude.
00046  * The error of the outer loop is computed using quaternion arithmetic, and thus
00047  * avoids gimbal locks as long as the attitude error is smaller than 90 degrees
00048  * on the pitch axis.
00049  *
00050  ******************************************************************************/

00085 template<class TRate_controller>
00086 void Attitude_controller<TRate_controller>::update()
00087 {
00088     /* check whether this is the highest active level of the cascade */
00089     if(TRate_controller::cascade_command_ == &att_command_)
00090     {
00091         /* calculate torque_command and propagate down the cascade */
00092         typename TRate_controller::rate_command_t rate_command = calc_rate_command(att_command_);
00093         TRate_controller::update_cascade(rate_command);
00094     }else
00095     {
00096         /* propagate update() down the cascade until reaching the highest active level */
00097         TRate_controller::update();
00098     }
00099 }

00124 typename TRate_controller::rate_command_t Attitude_controller<TRate_controller>::calc_rate_command(const att_command_t& att_command)
00125 {
00126     float now      = time_keeper_get_s();
00127     dt_s_          = now - last_update_s_;
00128     last_update_s_ = now;
00129 
00130     float errors[3];
00131     typename TRate_controller::rate_command_t rate_command;
00132 
00133     // Get attitude command
00134     attitude_error_estimator_set_quat_ref(&attitude_error_estimator_,
00135                                           att_command.att);
00136 
00137     // Get local angular errors
00138     attitude_error_estimator_update(&attitude_error_estimator_);
00139     errors[ROLL]    = attitude_error_estimator_.rpy_errors[ROLL];
00140     errors[PITCH]   = attitude_error_estimator_.rpy_errors[PITCH];
00141     errors[YAW]     = attitude_error_estimator_.rpy_errors[YAW];
00142 
00143     // Update PIDs
00144     rate_command.rates[ROLL]  = pid_controller_update_dt(&pid_[ROLL],  errors[ROLL],  dt_s_);
00145     rate_command.rates[PITCH] = pid_controller_update_dt(&pid_[PITCH], errors[PITCH], dt_s_);
00146     rate_command.rates[YAW]   = pid_controller_update_dt(&pid_[YAW],   errors[YAW],   dt_s_);
00147 
00148     rate_command.thrust = att_command.thrust;
00149     return rate_command;
