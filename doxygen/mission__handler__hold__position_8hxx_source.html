00065 template <class T>
00066 bool Mission_handler_hold_position<T>::can_handle(const Waypoint& wpt) const
00067 {
00068     bool handleable = false;
00069 
00070     uint16_t cmd = wpt.command();
00071     if (cmd == MAV_CMD_NAV_LOITER_UNLIM ||
00072         cmd == MAV_CMD_NAV_LOITER_TIME  ||
00073         cmd == MAV_CMD_NAV_LOITER_TO_ALT ||
00074         (cmd == MAV_CMD_OVERRIDE_GOTO && wpt.param1() == MAV_GOTO_DO_HOLD))
00075     {
00076         handleable = true;
00077     }
00078 
00079     return handleable;
00080 }

00093 template <class T>
00094 Mission_handler::update_status_t Mission_handler_hold_position<T>::update()
00095 {
00096     // Set goal
00097     bool ret = set_control_command();
00098 
00099     /*******************
00100     Determine status code 
00101     ********************/
00102     // Determine if we have entered the hold position volume
00103     local_position_t wpt_pos = waypoint_.local_pos();
00104 
00105     float radius;
00106     if (!waypoint_.radius(radius))
00107     {
00108         radius = 0.0f;
00109     }
00110     
00111     // Check that we are at the waypoint
00112     else if (((wpt_pos[X]-ins_.position_lf()[X])*(wpt_pos[X]-ins_.position_lf()[X]) + (wpt_pos[Y]-ins_.position_lf()[Y])*(wpt_pos[Y]-ins_.position_lf()[Y])) < radius*radius)
00113     {
00114         within_radius_ = true;
00115     }
00116     else
00117     {
00118         within_radius_ = false;
00119 
00120         // Reset start time
00121         start_time_ = time_keeper_get_ms();
00122     }
00123 
00124     // Determine if we should advance to the next waypoint
00125     if (waypoint_.autocontinue() == 1)
00126     {
00127         switch (waypoint_.command())
00128         {
00129         case MAV_CMD_NAV_LOITER_UNLIM:
00130             break;
00131 
00132         case MAV_CMD_NAV_LOITER_TIME:
00133             if (within_radius_ && ((time_keeper_get_ms() - start_time_) > waypoint_.param1() * 1000))
00134             {
00135                 return MISSION_FINISHED;
00136             }
00137             break;
00138 
00139         case MAV_CMD_NAV_LOITER_TO_ALT:
00140             if (maths_f_abs(ins_.position_lf()[Z] - waypoint_.local_pos()[Z]) < waypoint_.param2()) // TODO: Add check for heading
00141             {
00142                 return MISSION_FINISHED;
00143             }
00144             break;
00145         case MAV_CMD_OVERRIDE_GOTO:
00146             return MISSION_FINISHED;
00147         }
00148     }
00149 
00150     // Handle control command failed status
00151     if (!ret)
00152     {
00153         return MISSION_FAILED;
00154     }
00155 
00156     return MISSION_IN_PROGRESS;
00157 }
