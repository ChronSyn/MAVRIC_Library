00064 template<class TPosition_controller>
00065 void Navigation_directto<TPosition_controller>::update()
00066 {
00067     /* check whether this is the highest active level of the cascade */
00068     if(TPosition_controller::cascade_command_ == &navigation_command_)
00069     {
00070         /* calculate pos_command and propagate down the cascade */
00071         pos_command_t pos_command = calc_position_command(navigation_command_);
00072         TPosition_controller::update_cascade(pos_command);
00073     }
00074     else
00075     {
00076         /* propagate update() down the cascade until reaching the highest active level */
00077         TPosition_controller::update();
00078     }
00079 }
