00121 template <class T>
00122 Mission_handler::update_status_t Mission_handler_navigating<T>::update()
00123 {
00124     // Set goal
00125     bool ret = set_control_command();
00126     
00127     /**********************************
00128     Determine if arrived for first time 
00129     **********************************/
00130     // Find distance to waypoint
00131     local_position_t wpt_pos = waypoint_.local_pos();
00132     float rel_pos[3];
00133     for (int i = 0; i < 3; i++)
00134     {
00135         rel_pos[i] = wpt_pos[i] - ins_.position_lf()[i];
00136     }
00137     float dist2wp_sqr = vectors_norm_sqr(rel_pos);
00138 
00139     // Check if radius is available
00140     float radius;
00141     if (!waypoint_.radius(radius))
00142     {
00143         radius = 4.0f;
00144     }
00145 
00146     // If we are near the waypoint or are doing dubin circles
00147     if (dist2wp_sqr < (radius * radius))
00148     {
00149         // If we are near the waypoint but the flag has not been set, do this once ...
00150         if (!waypoint_reached_)
00151         {
00152             // Send debug log ...
00153             print_util_dbg_print("Waypoint reached, distance: ");
00154             print_util_dbg_putfloat(sqrt(dist2wp_sqr), 3);
00155             print_util_dbg_print(" m. Less than acceptable radius:");
00156             print_util_dbg_putfloat(sqrt(radius * radius), 3);
00157             print_util_dbg_print(" m.\r\n");
00158 
00159             // ... as well as a mavlink message ...
00160             mavlink_message_t msg;
00161             mavlink_msg_mission_item_reached_pack(mavlink_stream_.sysid(),
00162                                                   mavlink_stream_.compid(),
00163                                                   &msg,
00164                                                   waypoint_handler_.current_waypoint_index());
00165             mavlink_stream_.send(&msg);
00166 
00167             // ... and record the travel time ...
00168             travel_time_ = time_keeper_get_ms() - start_time_;
00169 
00170             // ... and set to waiting at waypoint ...
00171             waypoint_reached_ = true;
00172         }
00173     }
00174 
00175     /*******************
00176     Determine status code 
00177     ********************/
00178     // First check if we have reached the waypoint
00179     if (waypoint_reached_)
00180     {
00181         // Then ensure that we are in autocontinue
00182         if ((waypoint_.autocontinue() == 1) && (waypoint_handler_.waypoint_count() > 1))
00183         {
00184             // Differentiate between dubin and direct to
00185             return MISSION_FINISHED;
00186         }
00187     }
00188 
00189     // Handle control command failed status
00190     if (!ret)
00191     {
00192         return MISSION_FAILED;
00193     }
00194     
00195     return MISSION_IN_PROGRESS;
00196 }
