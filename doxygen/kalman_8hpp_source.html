00126 template<uint32_t n, uint32_t m, typename T>
00127 static void update(Mat<n,1,T>& x,
00128                    Mat<n,n,T>& P,
00129                    Mat<m,1,T>& z,
00130                    const Mat<m,n,T>& H,  
00131                    const Mat<m,m,T>& R,  
00132                    Mat<m,m,T>& S,  
00133                    Mat<n,m,T>& K,  
00134                    Mat<m,1,T>& y,
00135                    Mat<n,n,T>& I)
00136 {
00137     // Innovation
00138     y = z - H % x;
00139 
00140     // Innovation covariance
00141     S = H % P % ~H + R;
00142 
00143     // Kalman gain
00144     bool inversible;
00145     K = P % ~H % S.inv(inversible);
00146 
00147     if(inversible)
00148     {
00149         // Update state
00150         x = x + K % y;
00151 
00152         // Update state covariance
00153         P = (I - K % H) % P;
00154     }
00155 };
