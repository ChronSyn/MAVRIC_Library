00039  * \brief A controller for rate control, to be use within a cascade controller structure
00040  *
00041  * TODO: update this text
00042  * \details It takes a command in attitude (roll/pitch/yaw or quaternion) as
00043  * input, and computes a torque command on roll pitch and yaw.
00044  * The inner PID loop controls the angular speed around the 3 axis. This inner
00045  * loop is fed by the outer PID loop which controls the attitude.
00046  * The error of the outer loop is computed using quaternion arithmetic, and thus
00047  * avoids gimbal locks as long as the attitude error is smaller than 90 degrees
00048  * on the pitch axis.
00049  *
00050  ******************************************************************************/

00110 template<class TServos_mix>
00111 typename TServos_mix::torq_command_t Rate_controller<TServos_mix>::calc_torque_command(const rate_command_t& rate_command)
00112 {
00113     float now      = time_keeper_get_s();
00114     dt_s_          = now - last_update_s_;
00115     last_update_s_ = now;
00116 
00117     float errors[3];
00118     typename TServos_mix::torq_command_t torque_command;
00119 
00120     // Get errors on rate
00121     errors[ROLL]  = rate_command.rates[ROLL]  - ahrs_.angular_speed[ROLL];
00122     errors[PITCH] = rate_command.rates[PITCH] - ahrs_.angular_speed[PITCH];
00123     errors[YAW]   = rate_command.rates[YAW]   - ahrs_.angular_speed[YAW];
00124 
00125     // Update PIDs
00126     torque_command.torq[ROLL]  = pid_controller_update_dt(&pid_[ROLL],  errors[ROLL],  dt_s_);
00127     torque_command.torq[PITCH] = pid_controller_update_dt(&pid_[PITCH], errors[PITCH], dt_s_);
00128     torque_command.torq[YAW]   = pid_controller_update_dt(&pid_[YAW],   errors[YAW],   dt_s_);
00129 
00130     torque_command.thrust = rate_command.thrust;
00131 
00132     return torque_command;
00133 }
