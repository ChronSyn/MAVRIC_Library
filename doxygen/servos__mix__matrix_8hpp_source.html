00136     virtual bool update(void)
00137     {
00138         // Mix commands into servo values
00139         Mat<N,1> servos_cmd = mix_ % Mat<6,1>({torque_command_.xyz[X],
00140                                                torque_command_.xyz[Y],

00142                                                thrust_command_.xyz[X],

00145 
00146         // Add trim

00149         // Clip between min and max actuator values
00150         servos_cmd.clip(min_, max_);

00153         for (uint32_t i=0; i<N; ++i)
00154         {
