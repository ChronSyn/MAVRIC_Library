596 static inline unsigned char usart_lin_get_data_length(volatile avr32_usart_t *usart)
597 {
598  if (usart->linmr & AVR32_USART_LINMR_DLM_MASK)
599  {
600  unsigned char data_length = 1 << ((usart->linir >> (AVR32_USART_LINIR_IDCHR_OFFSET + 4)) & 0x03);

602  data_length = 2;
603  return data_length;
604  }
605  else
606  return ((usart->linmr & AVR32_USART_LINMR_DLC_MASK) >> AVR32_USART_LINMR_DLC_OFFSET) + 1;
607 }
