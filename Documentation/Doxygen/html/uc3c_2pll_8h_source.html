141 static inline void pll_config_init(struct pll_config *cfg,
142  enum pll_source src, unsigned int div, unsigned int mul)
143 {
144  uint32_t vco_hz;
145 
146  Assert(src < PLL_NR_SOURCES);
147 
148  /* Calculate internal VCO frequency */
149  vco_hz = osc_get_rate(src) * mul;
150  vco_hz /= div;
151  Assert(vco_hz >= PLL_MIN_HZ);
152  Assert(vco_hz <= PLL_MAX_HZ);
153 
154  cfg->ctrl = 0;
155 
156  /* Bring the internal VCO frequency up to the minimum value */
157  if ((vco_hz < PLL_MIN_HZ * 2) && (mul <= 8)) {
158  mul *= 2;
159  vco_hz *= 2;
160  pll_config_set_option(cfg, PLL_OPT_OUTPUT_DIV);
161  }
162 
163  /* Set VCO frequency range according to calculated value */
164  if (vco_hz < PLL_VCO_LOW_THRESHOLD)
165  pll_config_set_option(cfg, PLL_OPT_VCO_RANGE_LOW);
166 
167  Assert(mul > 2 && mul <= 16);
168  Assert(div > 0 && div <= 15);
169 
170  cfg->ctrl |= ((mul - 1) << AVR32_SCIF_PLLMUL)
171  | (div << AVR32_SCIF_PLLDIV)
172  | (PLL_MAX_STARTUP_CYCLES << AVR32_SCIF_PLLCOUNT)
173  | (src << AVR32_SCIF_PLLOSC);
174 }
