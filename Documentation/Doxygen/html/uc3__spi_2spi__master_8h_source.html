105 static inline void spi_master_init(volatile avr32_spi_t *spi)
106 {
107 #ifdef AVR32_SPI
108  sysclk_enable_pba_module(SYSCLK_SPI);
109 #else
110  if ((uint32_t)spi == AVR32_SPI0_ADDRESS)
111  {
112  sysclk_enable_pba_module(SYSCLK_SPI0);
113  }
114 #ifdef AVR32_SPI1
115  else if ((uint32_t)spi == AVR32_SPI1_ADDRESS)
116  {
117  sysclk_enable_pba_module(SYSCLK_SPI1);
118  }
119 #endif
120 #endif
121 
122  spi_reset(spi);
123  spi_set_master_mode(spi);
124  spi_disable_modfault(spi);
125  spi_disable_loopback(spi);
126  spi_set_chipselect(spi,(1 << AVR32_SPI_MR_PCS_SIZE) - 1);
127  spi_disable_variable_chipselect(spi);
128  spi_disable_chipselect_decoding(spi);
129  spi_set_delay(spi,CONFIG_SPI_MASTER_DELAY_BCS);
130 }
