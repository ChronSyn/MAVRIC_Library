77 {
78  // USART options.
79  opt->channelmode = CONFIG_USART_SERIAL_MODE;
80 #if (defined(AVR32_USART0_ADDRESS))

94  }
95 #endif
96 #if (defined(AVR32_USART2_ADDRESS))
97  if ((uint32_t)usart == AVR32_USART2_ADDRESS)

104  {
105  sysclk_enable_pba_module(SYSCLK_USART3);
106  }
107 #endif
108  usart_init_rs232(usart, opt, sysclk_get_pba_hz());
